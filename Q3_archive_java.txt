public class Q3 {
    public static void main(String[] args) {
        if (args.length != 1) {
            System.out.println("Usage: java Q3 'nombre'");
            return;
        }
        try {
            int N = Integer.parseInt(args[0]);
            for (int i = 1; i <= N; i++) {
                System.out.print(i+ " ");
            }
        } catch (NumberFormatException e) {
            System.out.println("Le paramètre N doit être un entier.");
        }
    }
}
